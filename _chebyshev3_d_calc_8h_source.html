177 inline Float_t Chebyshev3DCalc::chebyshevEvaluation1D(Float_t x, const Float_t* array, int ncf)
178 {
179  if (ncf <= 0) {
180  return 0;
181  }
182 
183  Float_t b0, b1, b2, x2 = x + x;
184  b0 = array[--ncf];
185  b1 = b2 = 0;
186 
187  for (int i = ncf; i--;) {
188  b2 = b1;
189  b1 = b0;
190  b0 = array[i] + x2 * b1 - b2;
191  }
192  return b0 - x * b1;
193 }

197 inline Float_t Chebyshev3DCalc::Eval(const Float_t* par) const
198 {
199  if (!mNumberOfRows) {
200  return 0.;
201  }
202  int ncfRC;
203  for (int id0 = mNumberOfRows; id0--;) {
204  int nCLoc = mNumberOfColumnsAtRow[id0]; // number of significant coefs on this row
205  int col0 = mColumnAtRowBeginning[id0]; // beginning of local column in the 2D boundary matrix
206  for (int id1 = nCLoc; id1--;) {
207  int id = id1 + col0;
208  mTemporaryCoefficients2D[id1] = (ncfRC = mCoefficientBound2D0[id])
209  ? chebyshevEvaluation1D(par[2], mCoefficients + mCoefficientBound2D1[id], ncfRC)
210  : 0.0;
211  }
212  mTemporaryCoefficients1D[id0] = nCLoc > 0 ? chebyshevEvaluation1D(par[1], mTemporaryCoefficients2D, nCLoc) : 0.0;
213  }
214  return chebyshevEvaluation1D(par[0], mTemporaryCoefficients1D, mNumberOfRows);
215 }
